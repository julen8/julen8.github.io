parted实现硬盘自动分区(GPT)

0.对磁盘进行分区

parted -s /dev/sda mklabel gpt
#这里只分成一个分区
parted -s /dev/sda mkpart primary 0% 100%

1.格式化分区

#选用exfat可以在Windows/linux/macos同时使用
mkfs.exfat /dev/sda1

2.挂载分区

mount /dev/sda1 /media/usb0

3.获取磁盘列表

fdisk -l | egrep -v mmc | grep -o "/dev/[a-z].*d[a-z]"

4.自动分区挂载脚本

#!/usr/bin/env python
# -*- coding: UTF-8 -*-
import getopt
import os
import sys


def run_cmd(cmd):
    raw_output = os.popen(cmd).readlines()
    output = []
    for v in raw_output:
        v = v.strip()  # 去除尾位空格及换行
        if v != '':
            output.append(v)
    return output


def get_disk_list():
    cmd = 'fdisk -l 2>/dev/null | egrep -v mmc | grep -o "/dev/[a-z].*d[a-z]"'
    return run_cmd(cmd)


class DiskFormat:
    def __init__(self, disk):
        self.disk = disk

    def get_info(self):
        cmd = 'parted -s ' + self.disk + ' print 2>&1'
        return os.popen(cmd).read()

    def make_partition(self):
        os.system('parted -s ' + self.disk + ' mklabel gpt')
        os.system('parted -s ' + self.disk + ' mkpart primary 0% 100%')

    def format_disk(self):
        os.system('mkfs.exfat %s1' % self.disk)

    def mount(self):
        dirs = run_cmd('ls /media/')
        for d in dirs:
            ret = run_cmd('mount | grep ' + d)
            if len(ret) > 0:
                continue
            cmd = 'mount %s1 /media/%s' % (self.disk, d)
            print(cmd)
            os.system(cmd)
            break

    def unmount(self):
        cmd = 'mount | grep %s | awk \'{print $1}\'' % self.disk
        ret = run_cmd(cmd)
        for i in ret:
            print('umount ' + i)
            os.system('umount ' + i)


def show_help():
    print(r'''Usage: 
    ''' + sys.argv[0] + r''' [OPTION]
    List of options:
        -h, --help          display this help and exit
        -l, --list          display disk list
        -i, --info=disk     display disk info, eg: -i /dev/sda
        -f, --format=disk   format disk, eg: -f /dev/sda
    ''')


def main(argv):
    opts = ''

    try:
        opts, args = getopt.gnu_getopt(argv, 'hli:f:', ['help', 'list', 'info=', 'format='])
    except getopt.GetoptError:
        show_help()
        exit(2)

    for opt, arg in opts:
        if opt in ('-h', '--help'):
            show_help()
            exit()
        elif opt in ('-l', '--list'):
            disks = get_disk_list()
            print(' '.join(disks))
            exit()
        elif opt in ('-i', '--info'):
            print(DiskFormat(arg).get_info())
            exit()
        elif opt in ('-f', '--format'):
            print('format %s :' % arg)
            disk_format = DiskFormat(arg)
            print('unmounting...')
            disk_format.unmount()
            print('making partition...')
            disk_format.make_partition()
            print('formatting disk...')
            disk_format.format_disk()
            print('mounting...')
            disk_format.mount()
            exit()

    show_help()
    quit()


if __name__ == '__main__':
    main(sys.argv[1:])
