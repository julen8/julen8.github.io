2.内核配置

  • config
Device Drivers  ‐‐‐>
    [*] SPI support  ‐‐‐>
        <*>   Rockchip SPI controller driver
        <*>   User mode SPI device driver support
  • dts
&spi1 {
    status = "okay";
    pinctrl-names = "default", "high_speed";
    pinctrl-0 = <&spi1m2_cs0 &spi1m2_pins>;
    pinctrl-1 = <&spi1m2_cs0 &spi1m2_pins_hs>;

    max-freq = <48000000>; // spi内部工作时钟
    dma-names = "tx","rx"; // 使能DMA模式，一般通讯字节少于32字节的不建议用
    spi_cs0 {
        status = "okay";
        compatible = "rockchip,spidev"; // 与驱动对应的名字    驱动文件 spidev.c
        reg = <0>; // 片选0或者1
        spi-max-frequency = <5000000>; // spi clk输出的时钟频率，不超过50M
        //spi‐cpha；                         // 如果有配，cpha为1
        //spi‐cpol；                         // 如果有配，cpol为1,clk脚保持高电平
        //spi‐cs‐high；                      // 如果有配，每传完一个数据，cs都会被拉高，再拉低
    };
};

#ifndef _SPI_H_INCLUDED
#define _SPI_H_INCLUDED


#include <linux/spi/spidev.h>
#include <cstdlib>
#include <cstdint>
#include <string>
#include <utility>


class SPIClass {
public:
    struct Setting {
        std::string device = "/dev/spidev0.0"; /* device to use*/
        uint32_t speed = 500000; /*max speed (Hz)*/
        uint16_t delay = 0; /*delay (usec)*/
        uint8_t bits = 8; /*bits per word*/
        uint32_t mode = 0; /* _spi mod eg: SPI_CPHA, SPI_LOOP ... */
    };

private:
    bool _isInit = false;
    bool _verbose = false;
    int _fd = -1;

public:
    Setting _spiSetting;

public:
    explicit SPIClass(Setting spiSetting, bool verbose = false);

    ~SPIClass();

    bool init();

    bool transfer(const uint8_t *data, uint8_t *out, uint32_t size);

    bool write(uint8_t *data, uint32_t count);

    bool write(uint8_t reg, uint8_t val);

    bool write(uint8_t reg, uint8_t *val, uint32_t count);

    // 注意:ret的第一个字节是无效数据
    // addZero: 末尾加零, 如果不加零的话
    bool read(uint8_t *data, uint32_t count, uint8_t *ret, bool addZero = true);

    bool read(uint8_t reg, uint8_t *ret);
};
#endif

#include <unistd.h>
#include <fcntl.h>
#include <linux/spi/spidev.h>
#include <sys/ioctl.h>
#include <cstring>
#include "SPI.h"
#include "debug.h"

#define SHOW_TX_RX 0

SPIClass::SPIClass(SPIClass::Setting spiSetting, bool verbose) : _spiSetting(std::move(spiSetting)), _verbose(verbose) {
}

SPIClass::~SPIClass() {
    if (_isInit && _fd > 0) {
        close(_fd);
        _isInit = false;
        _fd = -1;
    }
}

bool SPIClass::init() {
    do {
        _fd = open(_spiSetting.device.c_str(), O_RDWR);
        if (_fd < 0) {
            break;
        }

        /*
        * _spi mode
        */
        auto ret = ioctl(_fd, SPI_IOC_WR_MODE32, &_spiSetting.mode);
        if (ret == -1) {
            ERROR("can't set _spi mode");
            break;
        }

        ret = ioctl(_fd, SPI_IOC_RD_MODE32, &_spiSetting.mode);
        if (ret == -1) {
            ERROR("can't get _spi mode");
            break;
        }

        /*
         * bits per word
         */
        ret = ioctl(_fd, SPI_IOC_WR_BITS_PER_WORD, &_spiSetting.bits);
        if (ret == -1) {
            ERROR("can't set bits per word");
            break;
        }

        ret = ioctl(_fd, SPI_IOC_RD_BITS_PER_WORD, &_spiSetting.bits);
        if (ret == -1) {
            ERROR("can't get bits per word");
            break;
        }

        /*
         * max speed hz
         */
        ret = ioctl(_fd, SPI_IOC_WR_MAX_SPEED_HZ, &_spiSetting.speed);
        if (ret == -1) {
            ERROR("can't set max speed hz");
            break;
        }

        ret = ioctl(_fd, SPI_IOC_RD_MAX_SPEED_HZ, &_spiSetting.speed);
        if (ret == -1) {
            ERROR("can't get max speed hz");
            break;
        }

        if (_verbose) {
            INFO("_spi mode: 0x%x\n", _spiSetting.mode);
            INFO("bits per word: %d\n", _spiSetting.bits);
            INFO("max speed: %d Hz (%d KHz)\n", _spiSetting.speed, _spiSetting.speed / 1000);
        }
        _isInit = true;
    } while (false);

    if (!_isInit && _fd > 0) {
        close(_fd);
        _fd = -1;
    }

    return _isInit;
}

bool SPIClass::transfer(const uint8_t *data, uint8_t *out, uint32_t size) {
    struct spi_ioc_transfer tr = {
            .tx_buf = (unsigned long) data,
            .rx_buf = (unsigned long) out,
            .len = size,
            .speed_hz = _spiSetting.speed,
            .delay_usecs = _spiSetting.delay,
            .bits_per_word = _spiSetting.bits,
    };

    if (_spiSetting.mode & SPI_TX_QUAD)
        tr.tx_nbits = 4;
    else if (_spiSetting.mode & SPI_TX_DUAL)
        tr.tx_nbits = 2;
    if (_spiSetting.mode & SPI_RX_QUAD)
        tr.rx_nbits = 4;
    else if (_spiSetting.mode & SPI_RX_DUAL)
        tr.rx_nbits = 2;
    if (!(_spiSetting.mode & SPI_LOOP)) {
        if (_spiSetting.mode & (SPI_TX_QUAD | SPI_TX_DUAL))
            tr.rx_buf = 0;
        else if (_spiSetting.mode & (SPI_RX_QUAD | SPI_RX_DUAL))
            tr.tx_buf = 0;
    }

    auto ret = ioctl(_fd, SPI_IOC_MESSAGE(1), &tr);
    if (ret < 1) {
        ERROR("can't send _spi message");
        return false;
    }

#if SHOW_TX_RX
    hexDump(data, size, "TX");
    hexDump(out, size, "RX");
#endif
    return true;
}

bool SPIClass::write(uint8_t *data, uint32_t count) {
    auto ret = (uint8_t *) alloca(count);
    return transfer(data, ret, count);
}

bool SPIClass::write(uint8_t reg, uint8_t val) {

    uint8_t data[] = {reg, val};
    return write(data, 2);
}

bool SPIClass::write(uint8_t reg, uint8_t *val, uint32_t count) {
    auto data = (uint8_t *) alloca(count + 1);
    data[0] = reg;
    memcpy(data + 1, val, count);
    return write(data, count + 1);
}

bool SPIClass::read(uint8_t *data, uint32_t count, uint8_t *ret, bool addZero) {
    if (addZero) {
        auto sendData = (uint8_t *) alloca(count + 1);
        auto retData = (uint8_t *) alloca(count + 1);

        memcpy(sendData, data, count);
        sendData[count] = 0;
        bzero(retData, count + 1);

        auto r = transfer(sendData, retData, count + 1);
        if (r) {
            memcpy(ret, retData + 1, count);
        }
        return r;
    } else {
        return transfer(data, ret, count);
    }
}

bool SPIClass::read(uint8_t reg, uint8_t *ret) {
    *ret = 0;
    uint8_t send[] = {reg, 0x00};
    uint8_t recv[2] = {0x0};
    auto r = transfer(send, recv, 2);
    if (r) {
        *ret = recv[1];
    }

    return r;
}
